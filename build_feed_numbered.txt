   154	def _detect_stale_caches(report: RunReport, now: datetime) -> List[str]:
   155	    """Record warnings for provider caches older than the configured threshold."""
   156
   157	    if feed_config.CACHE_MAX_AGE_HOURS <= 0:
   158	        return []
   159
   160	    threshold = timedelta(hours=feed_config.CACHE_MAX_AGE_HOURS)
   161	    stale_messages: List[str] = []
   162
   163	    for _, loader in PROVIDERS:
   164	        cache_name = getattr(loader, "_provider_cache_name", None)
   165	        if not cache_name:
   166	            continue
   167
   168	        modified_at = cache_modified_at(str(cache_name))
   169	        if modified_at is None:
   170	            continue
   171
   172	        if modified_at.tzinfo is None:
   173	            modified_at = modified_at.replace(tzinfo=timezone.utc)
   174
   175	        age = now - modified_at
   176	        if age <= threshold:
   177	            continue
   178
   179	        hours = age.total_seconds() / 3600
   180	        message = (
   181	            f"Cache {cache_name}: zuletzt vor {hours:.1f}h aktualisiert "
   182	            f"(Schwelle {feed_config.CACHE_MAX_AGE_HOURS}h)"
   183	        )
   184	        report.add_warning(message)
   185	        stale_messages.append(message)
   186
   187	    return stale_messages

   212	def _validate_configuration(statuses: List[Tuple[str, bool]]) -> None:
   213	    """Check the runtime configuration for common issues (e.g. no providers active)."""
   214	    enabled_count = sum(1 for _, is_enabled in statuses if is_enabled)
   215	    if not statuses:
   216	        log.warning("Keine Provider registriert – es werden keine Items gesammelt.")
   217	    elif enabled_count == 0:
   218	        log.error(
   219	            "Alle Provider deaktiviert – Feed bleibt leer, bitte Konfiguration prüfen."
   220	        )
   221
   222	    if feed_config.MAX_ITEMS == 0:
   223	        log.warning("MAX_ITEMS ist 0 – der Feed wird ohne Einträge erzeugt.")
   224	    if feed_config.FEED_TTL == 0:
   225	        log.warning(
   226	            "FEED_TTL ist 0 – Clients werten den Feed unmittelbar als abgelaufen."
   227	        )
   228	    if feed_config.PROVIDER_TIMEOUT == 0 and enabled_count:
   229	        log.warning(
   230	            "PROVIDER_TIMEOUT ist 0 – Netzwerkprovider haben keine Zeit für Antworten."
   231	        )
   232	    if feed_config.MAX_ITEM_AGE_DAYS > feed_config.ABSOLUTE_MAX_AGE_DAYS:
   233	        log.warning(
   234	            "MAX_ITEM_AGE_DAYS (%s) übersteigt ABSOLUTE_MAX_AGE_DAYS (%s) – ältere Items "

   248	def _read_optional_non_negative_int(env_name: str) -> Optional[int]:
   249	    """Read a non-negative integer from an environment variable."""
   250	    raw = os.getenv(env_name)
   251	    if raw is None:
   252	        return None
   253	    stripped = raw.strip()
   254	    if not stripped:
   255	        return None
   256	    try:
   257	        value = int(stripped)
   258	    except (TypeError, ValueError) as exc:
   259	        log.warning(
   260	            "Ungültiger Wert für %s=%r – ignoriere Override (%s: %s)",
   261	            env_name,
   262	            raw,
   263	            type(exc).__name__,
   264	            exc,
   265	        )
   266	        return None
   267	    if value < 0:
   268	        log.warning("Negativer Wert für %s=%r – ignoriere Override", env_name, raw)
   269	        return None
   270	    return value

   336	def _fetch_supports_timeout(fetch: Any) -> bool:
   337	    """Check if the fetch callable accepts a 'timeout' argument."""
   338	    try:
   339	        signature = inspect.signature(fetch)
   340	    except (TypeError, ValueError):
   341	        return False
   342	    for param in signature.parameters.values():
   343	        if param.kind == inspect.Parameter.VAR_KEYWORD:
   344	            return True
   345	        if param.name == "timeout":
   346	            return True
   347	    return False

   363	def _to_utc(dt: datetime) -> datetime:
   364	    """Return a timezone-aware datetime in UTC.
   365
   366	    Strictly requires timezone-aware datetimes.
   367	    """
   368	    if dt.tzinfo is None:
   369	        raise ValueError("Naive datetimes are banned. Must be timezone-aware.")
   370
   371	    if dt.tzinfo is timezone.utc:
   372	        return dt
   373
   374	    return dt.astimezone(timezone.utc)

   376	def _fmt_rfc2822(dt: datetime) -> str:
   377	    """Format datetime as RFC-2822 string with Vienna offset."""
   378	    try:
   379	        # Convert to Vienna time for output
   380	        local_dt = _to_utc(dt).astimezone(_VIENNA_TZ)
   381	        return format_datetime(local_dt)
   382	    except Exception:
   383	        log.exception(
   384	            "Konnte Datum %r nicht per format_datetime formatieren – nutze strftime-Fallback.",
   385	            dt,
   386	        )
   387	        try:
   388	            local_dt = _to_utc(dt).astimezone(_VIENNA_TZ)
   389	        except Exception:
   390	            if dt.tzinfo is None:
   391	                local_dt = dt.replace(tzinfo=timezone.utc)
   392	            else:
   393	                local_dt = dt.astimezone(timezone.utc)
   394	        days = ["Mon", "Tue", "Wed", "Thu", "Fri", "Sat", "Sun"]
   395	        months = ["Jan", "Feb", "Mar", "Apr", "May", "Jun", "Jul", "Aug", "Sep", "Oct", "Nov", "Dec"]
   396
   397	        day_name = days[local_dt.weekday()]
   398	        month_name = months[local_dt.month - 1]
   399
   400	        offset = local_dt.utcoffset()
   401	        if offset is None:
   402	            offset_str = "-0000"
   403	        else:
   404	            total_seconds = int(offset.total_seconds())
   405	            hours, remainder = divmod(abs(total_seconds), 3600)
   406	            minutes, _ = divmod(remainder, 60)
   407	            sign = "+" if total_seconds >= 0 else "-"
   408	            offset_str = f"{sign}{hours:02d}{minutes:02d}"
   409
   410	        return (
   411	            f"{day_name}, {local_dt.day:02d} {month_name} {local_dt.year:04d} "
   412	            f"{local_dt.hour:02d}:{local_dt.minute:02d}:{local_dt.second:02d} {offset_str}"
   413	        )

   416	def format_local_times(
   417	    start: Optional[datetime], end: Optional[datetime]
   418	) -> str:
   419	    """Format a time range (start, end) into a localized string (e.g. 'Seit 01.01.2023')."""
   420	    start_local: Optional[datetime] = None
   421	    end_local: Optional[datetime] = None
   422
   423	    if isinstance(start, datetime):
   424	        start_local = _to_utc(start).astimezone(_VIENNA_TZ)
   425	    if isinstance(end, datetime):
   426	        end_local = _to_utc(end).astimezone(_VIENNA_TZ)
   427
   428	    if start_local and end_local and (end_local - start_local).days > 180:
   429	        end_local = None
   430
   431	    today = datetime.now(_VIENNA_TZ)
   432
   433	    if start_local:
   434	        if end_local:
   435	            if start_local.date() == end_local.date():
   436	                return f"Am {start_local:%d.%m.%Y}"
   437	            if end_local < start_local:
   438	                log.warning("Enddatum liegt vor Startdatum")
   439	                if start_local.date() > today.date():
   440	                    return f"Ab {start_local:%d.%m.%Y}"
   441	                return f"Seit {start_local:%d.%m.%Y}"
   442	            return f"{start_local:%d.%m.%Y} – {end_local:%d.%m.%Y}"
   443	        if start_local.date() > today.date():
   444	            return f"Ab {start_local:%d.%m.%Y}"
   445	        return f"Seit {start_local:%d.%m.%Y}"
   446	    if end_local:
   447	        return f"bis {end_local:%d.%m.%Y}"
   448	    return ""

   470	def _parse_lines_from_title(title: str) -> List[str]:
   471	    m = _LINE_PREFIX_RE.match(title or "")
   472	    if not m:
   473	        return []
   474
   475	    tokens: List[str] = []
   476	    for raw in m.group(1).split("/"):
   477	        token = raw.strip()
   478	        if not token:
   479	            continue
   480	        normalized = token.upper()
   481	        if _LINE_TOKEN_RE.match(normalized):
   482	            tokens.append(normalized)
   483	    return tokens

   491	def _parse_datetime(value: Any) -> Optional[datetime]:
   492	    """Parse ISO8601 timestamps (incl. ``Z`` suffix and compact offsets).
   493
   494	    Treats naive timestamps as Vienna local time.
   495	    """
   496
   497	    if isinstance(value, datetime):
   498	        if value.tzinfo:
   499	            return value
   500	        return value.replace(tzinfo=_VIENNA_TZ)
   501
   502	    if isinstance(value, str):
   503	        candidate = value.strip()
   504	        if not candidate:
   505	            return None
   506
   507	        try:
   508	            parsed = parser.isoparse(candidate)
   509	            if parsed.tzinfo is None:
   510	                # Assume Vienna time for naive strings (e.g. from legacy cache)
   511	                parsed = parsed.replace(tzinfo=_VIENNA_TZ)
   512	            return parsed
   513	        except (ValueError, parser.ParserError) as exc:
   514	            log.debug("Datetime-Parsing fehlgeschlagen für %r (%s)", value, exc)
   515
   516	    return None

   519	def _coerce_datetime_field(it: Dict[str, Any], field: str) -> Optional[datetime]:
   520	    value = it.get(field)
   521	    if value is None:
   522	        return None
   523
   524	    parsed = _parse_datetime(value)
   525	    if parsed is None:
   526	        if isinstance(value, str):
   527	            log.warning("%s Parsefehler: %r", field, value)
   528	        it[field] = None
   529	        return None
   530
   531	    it[field] = parsed
   532	    return parsed

   548	def _load_state() -> Dict[str, Dict[str, Any]]:
   549	    path = validate_path(feed_config.STATE_FILE, "STATE_PATH")
   550	    try:
   551	        lock_path = path.with_suffix(".lock")
   552	        with lock_path.open("a+", encoding="utf-8") as lock_file:
   553	            with file_lock(lock_file, exclusive=False):
   554	                with path.open("r", encoding="utf-8") as f:
   555	                    data = json.load(f)
   556	        data = data if isinstance(data, dict) else {}
   557	    except (FileNotFoundError, json.JSONDecodeError):
   558	        return {}
   559	    except Exception as e:
   560	        log.warning("State laden fehlgeschlagen (%s) – starte leer.", e)
   561	        return {}
   562
   563	    retention_cutoff: Optional[datetime] = None
   564	    if feed_config.STATE_RETENTION_DAYS > 0:
   565	        now_utc = _to_utc(datetime.now(timezone.utc))
   566	        retention_cutoff = now_utc - timedelta(days=feed_config.STATE_RETENTION_DAYS)
   567
   568	    out: Dict[str, Dict[str, Any]] = {}
   569	    for ident, entry in data.items():
   570	        if not isinstance(entry, dict):
   571	            continue
   572	        try:
   573	            raw_first_seen = entry.get("first_seen", "")
   574	            fs_dt = datetime.fromisoformat(str(raw_first_seen))
   575	            if fs_dt.tzinfo is None:
   576	                fs_dt = fs_dt.replace(tzinfo=timezone.utc)
   577	            fs_utc = _to_utc(fs_dt)
   578	        except Exception:
   579	            log.warning(
   580	                "State-Eintrag %s hat unparsebares first_seen: %r", ident, entry.get("first_seen")
   581	            )
   582	            continue

   584	        if retention_cutoff and fs_utc < retention_cutoff:
   585	            log.debug(
   586	                "State-Eintrag %s älter als %s Tage – entferne Eintrag.",
   587	                ident,
   588	                feed_config.STATE_RETENTION_DAYS,
   589	            )
   590	            continue
   591
   592	        entry["first_seen"] = fs_utc.isoformat()
   593	        out[ident] = entry
   594	    return out

   596	def _save_state(state: Dict[str, Dict[str, Any]], deletions: Optional[Set[str]] = None) -> None:
   597	    path = validate_path(feed_config.STATE_FILE, "STATE_PATH")
   598	    path.parent.mkdir(parents=True, exist_ok=True)
   599	    # Windows-fix: Use a separate lock file to avoid permission errors when atomic_write replaces the target
   600	    lock_path = path.with_suffix(".lock")
   601	    with lock_path.open("a+", encoding="utf-8") as lock_file:
   602	        with file_lock(lock_file, exclusive=True):
   603	            # Safe merge: read existing state to avoid overwriting parallel updates
   604	            merged_state = {}
   605	            try:
   606	                with path.open("r", encoding="utf-8") as f:
   607	                    existing = json.load(f)
   608	                    if isinstance(existing, dict):
   609	                        merged_state = existing
   610	            except FileNotFoundError:
   611	                pass
   612	            except Exception as exc:
   613	                log.warning(
   614	                    "State-Merge fehlgeschlagen (Lesefehler: %s) – überschreibe State.",
   615	                    exc,
   616	                )
   617
   618	            merged_state.update(state)
   619	            if deletions:
   620	                for k in deletions:
   621	                    merged_state.pop(k, None)
   622
   623	            with atomic_write(
   624	                path, mode="w", encoding="utf-8", permissions=0o600
   625	            ) as f:
   626	                json.dump(merged_state, f, ensure_ascii=False, indent=2, sort_keys=True)

   628	def _identity_for_item(item: FeedItem) -> str:
   629	    """
   630	    Stabile Identität unabhängig von Titel-Kosmetik.
   631	      - Wenn Provider _identity liefert: diese bevorzugen.
   632	      - ÖBB: GUID/Link (vom RSS stabil).
   633	      - WL/sonstige: Quelle|Kategorie|Linienpräfix + Start-YYYY-MM-DD.
   634	    """
   635	    if item.get("_identity"):
   636	        return str(item["_identity"])
   637
   638	    if "_calculated_identity" in item:
   639	        return cast(str, item["_calculated_identity"])
   640
   641	    title = item.get("title") or ""
   642	    sa = item.get("starts_at")
   643	    ea = item.get("ends_at")
   644	    sa_str = _to_utc(sa).isoformat() if isinstance(sa, datetime) else "None"
   645	    ea_str = _to_utc(ea).isoformat() if isinstance(ea, datetime) else "None"
   646	    fuzzy_raw = f"{title}|{sa_str}|{ea_str}"
   647	    fuzzy_hash = hashlib.sha256(fuzzy_raw.encode("utf-8")).hexdigest()
   648
   649	    result: str
   650	    source = (item.get("source") or "").lower()
   651	    category = (item.get("category") or "").lower()
   652	    if "öbb" in source or "oebb" in source:
   653	        guid_or_link = item.get("guid") or item.get("link")
   654	        if guid_or_link:
   655	            result = f"oebb|{guid_or_link}"
   656	        else:
   657	            result = f"oebb|F={fuzzy_hash}"
   658	    else:
   659	        lines = _parse_lines_from_title(title)
   660	        lines_part = "L=" + "/".join(lines) if lines else "L="
   661	        start_day = _ymd_or_none(sa)
   662	        base = f"{source}|{category}|{lines_part}|D={start_day}"
   663	        if source and category:
   664	            if not lines:
   665	                if item.get("title"):
   666	                    result = f"{base}|T={item['title']}|F={fuzzy_hash}"
   667	                else:
   668	                    raw = json.dumps(item, sort_keys=True, default=str)
   669	                    hashed = hashlib.sha256(raw.encode("utf-8")).hexdigest()
   670	                    result = f"{base}|H={hashed}|F={fuzzy_hash}"
   671	            else:
   672	                result = f"{base}|F={fuzzy_hash}"
   673	        # Fallback: Ohne Quelle/Kategorie Titel oder vollständigen Hash anhängen
   674	        elif item.get("title"):
   675	            result = f"{base}|T={item['title']}|F={fuzzy_hash}"
   676	        else:
   677	            raw = json.dumps(item, sort_keys=True, default=str)
   678	            hashed = hashlib.sha256(raw.encode("utf-8")).hexdigest()
   679	            result = f"{base}|H={hashed}|F={fuzzy_hash}"
   680
   681	    item["_calculated_identity"] = result
   682	    return result

   686	def _collect_items(report: Optional[RunReport] = None) -> List[FeedItem]:
   687	    if report is None:
   688	        report = RunReport(provider_statuses())
   689	    items: List[FeedItem] = []
   690
   691	    cache_alerts: defaultdict[str, List[str]] = defaultdict(list)
   692	    seen_cache_alerts: set[Tuple[str, str]] = set()
   693	    alert_lock = Lock()
   694
   695	    def _cache_alert_handler(provider_key: str, message: str) -> None:
   696	        normalized_key = str(provider_key or "").strip()
   697	        normalized_message = clean_message(message)
   698	        if not normalized_key or not normalized_message:
   699	            return
   700	        with alert_lock:
   701	            cache_alerts[normalized_key].append(normalized_message)
   702	            if report is not None:
   703	                key = (normalized_key, normalized_message)
   704	                if key not in seen_cache_alerts:
   705	                    seen_cache_alerts.add(key)
   706	                    report.add_warning(f"Cache {normalized_key}: {normalized_message}")
   707
   708	    unregister_cache_alert = register_cache_alert_hook(_cache_alert_handler)
   709	    try:
   710	        cache_fetchers: List[Any] = []
   711	        network_fetchers: List[Any] = []
   712	        provider_names: Dict[Any, str] = {}
   713	        provider_envs: Dict[Any, Optional[str]] = {}
   714
   715	        provider_entries = list(PROVIDERS)
   716	        providers_overridden = tuple(PROVIDERS) != DEFAULT_PROVIDERS
   717	        if provider_entries:
   718	            if not providers_overridden:
   719	                known_envs = {env for env, _ in provider_entries}
   720	                for spec in iter_providers():
   721	                    if spec.env_var not in known_envs:
   722	                        provider_entries.append((spec.env_var, spec.loader))
   723	        else:
   724	            provider_entries = [(spec.env_var, spec.loader) for spec in iter_providers()]
   725
   726	        for env, fetch in provider_entries:
   727	            provider_name = _provider_display_name(fetch, env)
   728	            enabled = bool(feed_config.get_bool_env(env, True))
   729	            fetch_type = "cache" if getattr(fetch, "_provider_cache_name", None) else "network"
   730	            report.register_provider(provider_name, enabled, fetch_type)
   731	            if not enabled:
   732	                continue
   733	            provider_names[fetch] = provider_name
   734	            provider_envs[fetch] = env
   735	            if getattr(fetch, "_provider_cache_name", None):
   736	                cache_fetchers.append(fetch)
   737	            else:
   738	                network_fetchers.append(fetch)
   739
   740	        if not cache_fetchers and not network_fetchers:
   741	            return []
   742
   743	        def _merge_result(fetch: Any, result: Any, provider_name: str) -> None:
   744	            name = getattr(fetch, "__name__", str(fetch))
   745	            if not isinstance(result, list):
   746	                log.error("%s fetch gab keine Liste zurück: %r", name, result)
   747	                report.provider_error(provider_name, "Ungültige Antwort (keine Liste)")
   748	                return
   749	            # Cast raw dicts to FeedItem for typing compliance after normalization
   750	            _normalize_item_datetimes(result)
   751	            typed_result = cast(List[FeedItem], result)
   752	            items.extend(typed_result)
   753	            count = len(result)
   754	            if count == 0:
   755	                log.warning(
   756	                    "Cache für Provider '%s' leer – generiere Feed ohne aktuelle Daten.",
   757	                    provider_name,
   758	                )
   759	                detail = "Keine aktuellen Daten"
   760	                cache_name = getattr(fetch, "_provider_cache_name", None)
   761	                if cache_name is not None:
   762	                    alerts = cache_alerts.get(str(cache_name), [])
   763	                    if alerts:
   764	                        unique_alerts = list(dict.fromkeys(alerts))
   765	                        detail = "; ".join(unique_alerts)
   766	                report.provider_success(
   767	                    provider_name,
   768	                    items=count,
   769	                    status="empty",
   770	                    detail=detail,
   771	                )
   772	                if detail:
   773	                    report.add_warning(f"Provider {provider_name}: {detail}")
   774	            else:
   775	                report.provider_success(provider_name, items=count)
   776
   777	        for fetch in cache_fetchers:
   778	            name = getattr(fetch, "__name__", str(fetch))
   779	            provider_name = provider_names.get(fetch, _provider_display_name(fetch))
   780	            report.provider_started(provider_name)
   781	            try:
   782	                result = fetch()
   783	            except Exception as exc:
   784	                log.exception("%s fetch fehlgeschlagen: %s", name, exc)
   785	                report.provider_error(provider_name, f"Fetch fehlgeschlagen: {exc}")
   786	                continue
   787	            _merge_result(fetch, result, provider_name)
   788
   789	        if not network_fetchers:
   790	            return items
   791
   792	        desired_workers = len(network_fetchers)
   793	        if feed_config.PROVIDER_MAX_WORKERS > 0:
   794	            if desired_workers > feed_config.PROVIDER_MAX_WORKERS:
   795	                log.debug(
   796	                    "Begrenze Provider-Threads von %s auf %s",
   797	                    desired_workers,
   798	                    feed_config.PROVIDER_MAX_WORKERS,
   799	                )
   800	            desired_workers = min(desired_workers, feed_config.PROVIDER_MAX_WORKERS)
   801	        with ThreadPoolExecutor(max_workers=max(1, desired_workers)) as executor:
   802	            try:
   803	                futures: Dict[Any, Tuple[Any, str, int]] = {}
   804	                deadlines: Dict[Any, Optional[float]] = {}
   805	                pending: set[Any] = set()
   806	                cancelled_futures: set[Any] = set()
   807	                semaphores: Dict[str, BoundedSemaphore] = {}
   808
   809	                for fetch in network_fetchers:
   810	                    provider_name = provider_names.get(fetch, _provider_display_name(fetch))
   811	                    env_name = provider_envs.get(fetch)
   812	                    timeout_override = _provider_timeout_override(fetch, env_name, provider_name)
   813	                    effective_timeout = (
   814	                        timeout_override if timeout_override is not None else feed_config.PROVIDER_TIMEOUT
   815	                    )
   816	                    concurrency_key = _provider_concurrency_key(fetch, provider_name)
   817	                    worker_limit = _provider_worker_limit(
   818	                        fetch, env_name, provider_name, concurrency_key
   819	                    )
   820	                    semaphore: Optional[BoundedSemaphore] = None
   821	                    if worker_limit is not None and worker_limit > 0:
   822	                        semaphore = semaphores.get(concurrency_key)
   823	                        if semaphore is None:
   824	                            semaphore = BoundedSemaphore(worker_limit)
   825	                            semaphores[concurrency_key] = semaphore
   826	                    if timeout_override is not None:
   827	                        log.debug(
   828	                            "Provider %s nutzt Timeout-Override von %ss",
   829	                            provider_name,
   830	                            timeout_override,
   831	                        )
   832	                    if worker_limit is not None and worker_limit > 0:
   833	                        log.debug(
   834	                            "Provider %s begrenzt Worker auf %s (Schlüssel %s)",
   835	                            provider_name,
   836	                            worker_limit,
   837	                            concurrency_key,
   838	                        )
   839	                    supports_timeout = _fetch_supports_timeout(fetch)
   840
   841	                    if effective_timeout == 0:
   842	                        report.provider_started(provider_name)
   843	                        name = getattr(fetch, "__name__", str(fetch))
   844	                        log.error("%s fetch Timeout nach 0s", name)
   845	                        report.provider_error(provider_name, "Timeout nach 0s")
   846	                        continue
   847
   848	                    def _run_fetch(
   849	                        fetch: Any = fetch,
   850	                        timeout_value: Union[int, float] = effective_timeout,
   851	                        supports: bool = supports_timeout,
   852	                        semaphore: Optional[BoundedSemaphore] = semaphore,
   853	                        provider_name: str = provider_name,
   854	                    ) -> Any:
   855	                        timeout_arg = timeout_value if timeout_value >= 0 else None
   856
   857	                        if semaphore is None:
   858	                            return _call_fetch_with_timeout(fetch, timeout_arg, supports)
   859
   860	                        # Prevent thread starvation by enforcing timeout on semaphore acquisition
   861	                        # This ensures that if the provider is deadlocked/overloaded, we don't block
   862	                        # the executor worker forever.
   863	                        start_wait = perf_counter()
   864	                        # If timeout_arg is None, acquire(timeout=None) will wait indefinitely.
   865	                        acquired = semaphore.acquire(timeout=timeout_arg)
   866	                        if not acquired:
   867	                            raise TimeoutError(f"Semaphore acquisition timed out after {timeout_value}s")
   868
   869	                        # Task 3: Subtract wait time from timeout
   870	                        try:
   871	                            elapsed = perf_counter() - start_wait
   872	                            remaining_timeout = timeout_arg - elapsed if timeout_arg is not None else None
   873
   874	                            if remaining_timeout is not None and remaining_timeout <= 0:
   875	                                raise TimeoutError(
   876	                                    f"Semaphore acquisition took {elapsed:.2f}s, no realistic time left for fetch (threshold: <= 0s)"
   877	                                )
   878
   879	                            return _call_fetch_with_timeout(fetch, remaining_timeout, supports)
   880	                        finally:
   881	                            semaphore.release()
   882
   883	                    report.provider_started(provider_name)
   884	                    future = executor.submit(_run_fetch)
   885	                    futures[future] = (fetch, provider_name, effective_timeout)
   886	                    pending.add(future)
   887	                    start_time = perf_counter()
   888	                    if effective_timeout > 0:
   889	                        deadlines[future] = start_time + effective_timeout
   890	                    else:
   891	                        deadlines[future] = None
   892
   893	                while pending:
   894	                    now = perf_counter()
   895	                    expired = []
   896	                    for future in list(pending):
   897	                        deadline = deadlines.get(future)
   898	                        if deadline is not None and now >= deadline:
   899	                            expired.append(future)
   900	                    for future in expired:
   901	                        pending.discard(future)
   902	                        fetch, provider_name, timeout_value = futures[future]
   903	                        name = getattr(fetch, "__name__", str(fetch))
   904	                        log.error("%s fetch Timeout nach %ss", name, timeout_value)
   905	                        report.provider_error(
   906	                            provider_name,
   907	                            f"Timeout nach {timeout_value}s",
   908	                        )
   909	                        future.cancel()
   910	                        cancelled_futures.add(future)
   911
   912	                    if not pending:
   913	                        break
   914
   915	                    wait_timeout: Optional[float] = None
   916	                    remaining = []
   917	                    for fut in pending:
   918	                        deadline = deadlines.get(fut)
   919	                        if deadline is not None:
   920	                            remaining.append(deadline - now)
   921	                    if remaining:
   922	                        wait_timeout = max(min(remaining), 0.1)
   923
   924	                    done, _ = wait(pending, timeout=wait_timeout, return_when=FIRST_COMPLETED)
   925	                    if not done:
   926	                        continue
   927
   928	                    for future in done:
   929	                        pending.discard(future)
   930	                        if future in cancelled_futures:
   931	                            continue
   932	                        fetch, provider_name, timeout_value = futures[future]
   933	                        name = getattr(fetch, "__name__", str(fetch))
   934	                        try:
   935	                            result = future.result()
   936	                        except (TimeoutError, requests.exceptions.Timeout) as exc:
   937	                            log.error("%s fetch Timeout: %s", name, exc)
   938	                            report.provider_error(provider_name, f"Timeout: {exc}")
   939	                        except CancelledError:
   940	                            report.provider_error(provider_name, "Fetch abgebrochen")
   941	                        except Exception as exc:
   942	                            log.exception("%s fetch fehlgeschlagen: %s", name, exc)
   943	                            report.provider_error(provider_name, f"Fetch fehlgeschlagen: {exc}")
   944	                        else:
   945	                            _merge_result(fetch, result, provider_name)
   946	            finally:
   947	                # Cancel remaining futures if we exit early or with exceptions.
   948	                # executor.shutdown(wait=False, cancel_futures=True) is automatically called by the context manager's __exit__,
   949	                # but we explicitly cancel pending futures to free resources immediately.
   950	                for future in pending:
   951	                    future.cancel()
   952
   953	        return items
   954	    finally:
   955	        unregister_cache_alert()

   962	def _drop_old_items(
   963	    items: List[FeedItem],
   964	    now: datetime,
   965	    state: Dict[str, Dict[str, Any]],
   966	) -> Tuple[List[FeedItem], Set[str]]:
   967	    """Entferne Items, die zu alt sind oder bereits beendet wurden.
   968
   969	    Neben ``pubDate``/``starts_at`` wird – falls vorhanden – ``first_seen`` aus dem
   970	    geladenen State als Altersreferenz verwendet. Das betrifft Items ohne
   971	    Datumsangaben, die andernfalls ewig im Feed verbleiben würden.
   972	    """
   973
   974	    out: List[FeedItem] = []
   975	    dropped: Set[str] = set()
   976	    now_utc = _to_utc(now)
   977	    for it in items:
   978	        if not isinstance(it, dict):
   979	            continue
   980
   981	        ident = _identity_for_item(it)
   982	        state_entry = state.get(ident) if isinstance(state, dict) else None
   983
   984	        ends_at = it.get("ends_at")
   985	        if isinstance(ends_at, datetime):
   986	            if _to_utc(ends_at) < now_utc - timedelta(minutes=feed_config.ENDS_AT_GRACE_MINUTES):
   987	                dropped.add(ident)
   988	                continue
   989
   990	        dt = it.get("pubDate") or it.get("starts_at")
   991	        age_days: Optional[float] = None
   992	        if isinstance(dt, datetime):
   993	            age_days = (now_utc - _to_utc(dt)).total_seconds() / 86400.0
   994	        elif isinstance(state_entry, dict):
   995	            raw_first_seen = state_entry.get("first_seen")
   996	            if raw_first_seen is not None:
   997	                try:
   998	                    first_seen_dt = datetime.fromisoformat(str(raw_first_seen))
   999	                except Exception:
  1000	                    log.warning(
  1001	                        "first_seen Parsefehler: %r – ignoriere für %s",
  1002	                        raw_first_seen,
  1003	                        ident,
  1004	                    )
  1005	                else:
  1006	                    if first_seen_dt.tzinfo is None:
  1007	                        first_seen_dt = first_seen_dt.replace(tzinfo=timezone.utc)
  1008	                    age_days = (now_utc - _to_utc(first_seen_dt)).total_seconds() / 86400.0
  1009
  1010	        if age_days is not None:
  1011	            if age_days > feed_config.ABSOLUTE_MAX_AGE_DAYS:
  1012	                dropped.add(ident)
  1013	                continue
  1014	            if age_days > feed_config.MAX_ITEM_AGE_DAYS:
  1015	                if not (
  1016	                    isinstance(ends_at, datetime) and _to_utc(ends_at) > now_utc
  1017	                ):
  1018	                    dropped.add(ident)
  1019	                    continue
  1020
  1021	        out.append(it)
  1022	    return out, dropped

  1025	def _dedupe_key_for_item(
  1026	    it: FeedItem, *, warn_on_missing: bool = True
  1027	) -> Tuple[str, bool]:
  1028	    """Return the deduplication key used for ``it`` and indicate fallback usage."""
  1029
  1030	    if it.get("_identity"):
  1031	        return str(it.get("_identity")), False
  1032
  1033	    if "_calculated_dedupe_key" in it:
  1034	        return cast(str, it["_calculated_dedupe_key"]), False
  1035
  1036	    guid = it.get("guid")
  1037	    source = (it.get("source") or "").lower()
  1038
  1039	    if "öbb" in source or "oebb" in source:
  1040	        guid_or_link = guid or it.get("link")
  1041	        if guid_or_link:
  1042	            return f"oebb|{guid_or_link}", False
  1043
  1044	    if guid:
  1045	        return str(guid), False
  1046	    raw = (
  1047	        f"{it.get('source') or ''}|{it.get('title') or ''}|{it.get('description') or ''}"
  1048	    )
  1049	    key = hashlib.sha256(raw.encode("utf-8")).hexdigest()
  1050
  1051	    # Only cache if we produced a fallback key to avoid recalculating SHA256
  1052	    # We set it before logging, so we can return it.
  1053	    it["_calculated_dedupe_key"] = key
  1054
  1055	    if warn_on_missing:
  1056	        log.warning(
  1057	            "Item ohne guid/_identity – Fallback-Schlüssel (source|title|description) %s",
  1058	            key,
  1059	        )
  1060	    return key, True

  1063	def _summarize_duplicates(items: Sequence[FeedItem]) -> List[DuplicateSummary]:
  1064	    groups: Dict[str, List[FeedItem]] = {}
  1065	    for it in items:
  1066	        if not isinstance(it, dict):
  1067	            continue
  1068	        key, _ = _dedupe_key_for_item(it, warn_on_missing=False)
  1069	        groups.setdefault(key, []).append(it)
  1070
  1071	    summaries: List[DuplicateSummary] = []
  1072	    for key, group in groups.items():
  1073	        if len(group) <= 1:
  1074	            continue
  1075	        titles = tuple(str(entry.get("title") or "") for entry in group[:3])
  1076	        summaries.append(
  1077	            DuplicateSummary(dedupe_key=key, count=len(group), titles=titles)
  1078	        )
  1079	    summaries.sort(key=lambda summary: summary.count, reverse=True)
  1080	    return summaries

  1098	def _dedupe_items(items: List[FeedItem]) -> List[FeedItem]:
  1099	    """
  1100	    Deduplicate items by identity/guid.
  1101
  1102	    When duplicates are found (same deduplication key), the "better" item is kept.
  1103	    Selection criteria for the "better" item:
  1104	    1. Later end date (indicates more up-to-date info on duration).
  1105	    2. More recent publication/start date.
  1106	    3. Longer description (indicates more detail).
  1107
  1108	    Args:
  1109	        items: List of item dictionaries.
  1110
  1111	    Returns:
  1112	        A list of unique item dictionaries.
  1113	    """
  1114
  1115	    def _recency_value(it: FeedItem) -> datetime:
  1116	        """Return a comparable timestamp describing how recent ``it`` is."""
  1117	        if "_calculated_recency" in it:
  1118	            return cast(datetime, it["_calculated_recency"])
  1119
  1120	        candidates: List[datetime] = []
  1121	        for field_name in ("pubDate", "first_seen", "starts_at"):
  1122	            value = it.get(field_name)  # type: ignore
  1123	            if isinstance(value, datetime):
  1124	                candidates.append(_to_utc(value))
  1125	            else:
  1126	                parsed = _parse_datetime(value)
  1127	                if isinstance(parsed, datetime):
  1128	                    candidates.append(_to_utc(parsed))
  1129
  1130	        if candidates:
  1131	            res = max(candidates)
  1132	        else:
  1133	            res = datetime.min.replace(tzinfo=timezone.utc)
  1134
  1135	        it["_calculated_recency"] = res
  1136	        return res
  1137
  1138	    def _end_value(it: FeedItem) -> datetime:
  1139	        if "_calculated_end" in it:
  1140	            return cast(datetime, it["_calculated_end"])
  1141
  1142	        ends = it.get("ends_at")
  1143	        if isinstance(ends, datetime):
  1144	            res = _to_utc(ends)
  1145	        else:
  1146	            res = datetime.min.replace(tzinfo=timezone.utc)
  1147
  1148	        it["_calculated_end"] = res
  1149	        return res
  1150
  1151	    def _better(a: FeedItem, b: FeedItem) -> bool:
  1152	        """Return True if ``a`` is better than ``b`` according to recency and content."""
  1153
  1154	        a_end = _end_value(a)
  1155	        b_end = _end_value(b)
  1156	        if a_end > b_end:
  1157	            return True
  1158	        if a_end < b_end:
  1159	            return False
  1160
  1161	        # Bei gleichem Enddatum: Zuerst Aktualität, dann Länge
  1162	        if _recency_value(a) > _recency_value(b):
  1163	            return True
  1164	        if _recency_value(a) < _recency_value(b):
  1165	            return False
  1166
  1167	        a_len = len(a.get("description") or "")
  1168	        b_len = len(b.get("description") or "")
  1169	        return a_len > b_len
  1170
  1171	    seen: Dict[str, int] = {}
  1172	    out: List[FeedItem] = []
  1173	    for it in items:
  1174	        key, _ = _dedupe_key_for_item(it)
  1175	        if key in seen:
  1176	            idx = seen[key]
  1177	            if _better(it, out[idx]):
  1178	                out[idx] = it
  1179	        else:
  1180	            seen[key] = len(out)
  1181	            out.append(it)
  1182	    return out

  1184	def _sort_key(item: FeedItem) -> Tuple[int, float, str]:
  1185	    pd = item.get("pubDate")
  1186	    # Fix TypeError: Ensure guid is always a string, even if explicitly None

  1189	    if isinstance(pd, datetime):
  1190	        return (0, -_to_utc(pd).timestamp(), guid_str)
  1191	    return (1, 0.0, guid_str)

  1220	def _emit_item(
  1221	    it: FeedItem, now: datetime, state: Dict[str, Dict[str, Any]]
  1222	) -> Tuple[str, ET.Element, Dict[str, str]]:
  1223	    """Convert a normalized item dictionary into an RSS <item> element and CDATA replacements.
  1224
  1225	    Args:
  1226	        it: The normalized item dictionary.
  1227	        now: The current datetime (used for relative time calculations).
  1228	        state: The state dictionary (used to persist first_seen timestamps).
  1229
  1230	    Returns:
  1231	        A tuple containing:
  1232	         - The item identity (str)
  1233	         - The generated ElementTree.Element
  1234	         - A dictionary mapping placeholder strings to their CDATA-wrapped content.
  1235	    """
  1236	    pubDate = _coerce_datetime_field(it, "pubDate")
  1237	    starts_at = _coerce_datetime_field(it, "starts_at")
  1238	    ends_at = _coerce_datetime_field(it, "ends_at")
  1239
  1240	    ident = _identity_for_item(it)
  1241	    st = state.get(ident)
  1242	    # Fallback: check guid as secondary key
  1243	    if not st and it.get("guid") and it["guid"] != ident:
  1244	        st = state.get(str(it["guid"]))
  1245	    if not st:
  1246	        st = {"first_seen": _to_utc(now).isoformat()}
  1247	    state[ident] = st
  1248
  1249	    try:
  1250	        fs_dt = datetime.fromisoformat(st["first_seen"])
  1251	    except Exception:
  1252	        log.warning("first_seen Parsefehler: %r – fallback to now", st.get("first_seen"))
  1253	        fs_dt = _to_utc(now)
  1254	        st["first_seen"] = fs_dt.isoformat()
  1255
  1256	    # Felder holen
  1257	    raw_title = it.get("title") or "Mitteilung"
  1258	    raw_desc  = it.get("description") or ""
  1259	    link = _build_canonical_link(it.get("link"), ident)
  1260	    sanitized_link = validate_http_url(link, check_dns=False) if link else ""
  1261	    if link and not sanitized_link:
  1262	        log.warning(
  1263	            "Item %s has potentially unsafe/invalid link %r; falling back to feed link.",
  1264	            ident,
  1265	            link,
  1266	        )
  1267	        link = ""
  1268	    else:
  1269	        link = sanitized_link
  1270
  1271	    if not link:
  1272	        link = feed_config.FEED_LINK
  1273
  1274	    raw_guid = it.get("guid") or ident
  1275	    guid = str(raw_guid).strip() if raw_guid is not None else ident
  1276	    if not guid:
  1277	        guid = ident
  1278	    if not isinstance(pubDate, datetime) and feed_config.FRESH_PUBDATE_WINDOW_MIN > 0:
  1279	        age = _to_utc(now) - _to_utc(fs_dt)
  1280	        if age <= timedelta(minutes=feed_config.FRESH_PUBDATE_WINDOW_MIN):
  1281	            pubDate = now
  1282
  1283	    # Task: Strict 2-line Layout (Summary + Timeframe)
  1284	    # Line 1: Concise plain text summary (no HTML artifacts)
  1285	    summary = html_to_text(raw_desc, collapse_newlines=True)
  1286	    summary = _sanitize_text(summary).strip()
  1287
  1288	    # ÖBB-spezifische Datumspräfixe (z.B. "17.09.2026 - 19.11.2026 • ") entfernen
  1289	    summary = re.sub(r"^\d{2}\.\d{2}\.\d{4}\s*-\s*\d{2}\.\d{2}\.\d{4}\s*•\s*", "", summary)
  1290	    summary = re.sub(r"^\d{2}\.\d{2}\.\d{4}\s*•\s*", "", summary)
  1291
  1292	    # Bulletpoints auflösen, um einen fließenden Satz zu bilden
  1293	    summary = summary.replace(" • ", " ").replace("•", " ")
  1294	    summary = _WHITESPACE_CLEANUP_RE.sub(" ", summary).strip()
  1295
  1296	    # Extrahiere maximal die ersten zwei Sätze
  1297	    sentences = [s.strip() for s in re.split(r'(?<=[.!?])\s+', summary) if s.strip()]
  1298	    if sentences:
  1299	        short_summary = sentences[0]
  1300	        # Falls der erste Satz sehr kurz ist, nimm den zweiten noch dazu
  1301	        if len(short_summary) < 60 and len(sentences) > 1:
  1302	            short_summary += " " + sentences[1]
  1303	        summary = short_summary
  1304
  1305	    # Harte Begrenzung für den TV-Screen (max. 180 Zeichen)
  1306	    if len(summary) > 180:
  1307	        summary = summary[:175].rsplit(' ', 1)[0] + " …"
  1308
  1309	    # Für XML robust aufbereiten (CDATA schützt Sonderzeichen)
  1310	    title_out = _sanitize_text(html.unescape(raw_title))
  1311	    if len(title_out) > feed_config.TITLE_CHAR_LIMIT:
  1312	        title_out = title_out[: feed_config.TITLE_CHAR_LIMIT].rstrip() + " …"
  1313
  1314	    # Minimal cleanup
  1315	    title_out = _WHITESPACE_RE.sub(" ", title_out).strip()
  1316	    title_cdata = _cdata_content(title_out)
  1317
  1318	    # Line 2: Timeframe
  1319	    time_line = format_local_times(
  1320	        starts_at if isinstance(starts_at, datetime) else None,
  1321	        ends_at if isinstance(ends_at, datetime) else None,
  1322	    )
  1323	    time_line = _sanitize_text(time_line)
  1324	    time_line = _WHITESPACE_CLEANUP_RE.sub(" ", time_line).strip()
  1325	    if time_line:
  1326	        if not time_line.startswith("["):
  1327	            time_line = f"[{time_line}"
  1328	        if not time_line.endswith("]"):
  1329	            time_line = f"{time_line}]"
  1330
  1331	    # Combine
  1332	    desc_parts = []
  1333	    if summary:
  1334	        desc_parts.append(summary)
  1335	    if time_line:
  1336	        desc_parts.append(time_line)
  1337
  1338	    desc_html = "<br/>".join(desc_parts)
  1339
  1340	    # Plain text summary for <description>
  1341	    desc_text = " ".join(desc_parts)
  1342	    desc_text_truncated = truncate_html(desc_text, feed_config.DESCRIPTION_CHAR_LIMIT, ellipsis="... [TRUNCATED]")
  1343
  1344	    # Truncate HTML descriptions using the HTML-aware truncator to prevent broken layout/XSS.
  1345	    desc_html = truncate_html(desc_html, feed_config.DESCRIPTION_CHAR_LIMIT, ellipsis="... [TRUNCATED]")
  1346
  1347	    # For title, we now rely on ElementTree's escaping which is safer/cleaner than manual CDATA.
  1348	    # ET will automatically escape <, >, & to &lt;, &gt;, &amp;.
  1349
  1350	    # Prepare CDATA content (handle ]]> in content)
  1351	    desc_cdata = _cdata_content(desc_html)
  1352
  1353	    # Generate unique placeholders
  1354	    # We use a cryptographically secure random token to ensure uniqueness within the document
  1355	    # Ensure placeholder is not accidentally present in the original desc_html or raw_desc
  1356	    while True:
  1357	        uid = secrets.token_hex(16)
  1358	        PH_CONTENT = f"___CDATA_CONTENT_{uid}___"
  1359	        PH_TITLE = f"___CDATA_TITLE_{uid}___"
  1360	        if PH_CONTENT not in desc_html and PH_CONTENT not in raw_desc and PH_TITLE not in title_out:
  1361	            break
  1362
  1363	    # --- ElementTree Construction ---
  1364	    item = ET.Element("item")
  1365
  1366	    # Title
  1367	    ET.SubElement(item, "title").text = PH_TITLE
  1368
  1369	    # Link
  1370	    ET.SubElement(item, "link").text = link
  1371
  1372	    # GUID
  1373	    guid_elem = ET.SubElement(item, "guid")
  1374	    guid_elem.text = guid
  1375
  1376	    # guid attributes (isPermaLink)
  1377	    parsed = urlparse(guid)
  1378	    if not (parsed.scheme and parsed.netloc and guid == link):
  1379	        guid_elem.set("isPermaLink", "false")
  1380
  1381	    # pubDate
  1382	    if isinstance(pubDate, datetime):
  1383	        ET.SubElement(item, "pubDate").text = _fmt_rfc2822(pubDate)
  1384
  1385	    # Extensions
  1386	    ET.SubElement(item, "{https://wien-oepnv.example/schema}first_seen").text = _fmt_rfc2822(fs_dt)
  1387
  1388	    if isinstance(starts_at, datetime):
  1389	        ET.SubElement(item, "{https://wien-oepnv.example/schema}starts_at").text = _fmt_rfc2822(starts_at)
  1390
  1391	    if isinstance(ends_at, datetime):
  1392	        ET.SubElement(item, "{https://wien-oepnv.example/schema}ends_at").text = _fmt_rfc2822(ends_at)
  1393
  1394	    # Description
  1395	    ET.SubElement(item, "description").text = desc_text_truncated
  1396
  1397	    # content:encoded
  1398	    ET.SubElement(item, "{http://purl.org/rss/1.0/modules/content/}encoded").text = PH_CONTENT
  1399
  1400	    replacements = {
  1401	        PH_CONTENT: f"<![CDATA[{desc_cdata}]]>",
  1402	        PH_TITLE: f"<![CDATA[{title_cdata}]]>",
  1403	    }
  1404
  1405	    return ident, item, replacements

  1408	def _make_rss(
  1409	    items: List[FeedItem],
  1410	    now: datetime,
  1411	    state: Dict[str, Dict[str, Any]],
  1412	    deletions: Optional[Set[str]] = None,
  1413	) -> Tuple[str, Optional[Set[str]]]:
  1414	    """
  1415	    Generate the full RSS XML document from a list of items using ElementTree.
  1416
  1417	    Args:
  1418	        items: List of item dictionaries.
  1419	        now: Current timestamp.
  1420	        state: State dictionary for tracking items.
  1421	        deletions: IDs to be removed from the state.
  1422
  1423	    Returns:
  1424	        A tuple containing:
  1425	        - The generated RSS XML string with CDATA sections.
  1426	        - The deletions set.
  1427	    """
  1428	    rss = ET.Element("rss", version="2.0")
  1429	    channel = ET.SubElement(rss, "channel")
  1430
  1431	    ET.SubElement(channel, "title").text = feed_config.FEED_TITLE
  1432	    ET.SubElement(channel, "link").text = feed_config.FEED_LINK
  1433	    ET.SubElement(channel, "description").text = feed_config.FEED_DESC
  1434	    ET.SubElement(channel, "lastBuildDate").text = _fmt_rfc2822(now)
  1435	    ET.SubElement(channel, "ttl").text = str(feed_config.FEED_TTL)
  1436
  1437	    item_replacements: Dict[str, str] = {}
  1438	    identities_in_feed: List[str] = []
  1439	    emitted = 0
  1440	    for it in items:
  1441	        if emitted >= feed_config.MAX_ITEMS:
  1442	            break
  1443	        ident, elem, repl = _emit_item(it, now, state)
  1444	        channel.append(elem)
  1445	        item_replacements.update(repl)
  1446	        identities_in_feed.append(ident)
  1447	        emitted += 1
  1448
  1449	    # Pretty print the tree
  1450	    if hasattr(ET, "indent"):
  1451	        ET.indent(rss, space="  ", level=0)
  1452
  1453	    # Serialize to string using native ElementTree declaration
  1454	    xml_bytes = ET.tostring(rss, encoding="utf-8", xml_declaration=True)
  1455	    xml_str = xml_bytes.decode("utf-8")
  1456
  1457	    # Inject CDATA
  1458	    for placeholder, cdata in item_replacements.items():
  1459	        xml_str = xml_str.replace(placeholder, cdata)
  1460
  1461	    return xml_str, deletions
